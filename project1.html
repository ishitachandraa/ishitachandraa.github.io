Project 1: The Mirror Of Erised

The Mirror of Erised project is an interactive, Harry Potter-inspired artifact using an Arduino. A photoresistor detects when a lit-up wand is waved in front of the mirror, triggering the servo to open a hidden door. A stone/container with a secret message is then revealed. On lifting up the container, LEDs light up in the background giving a magical glow. This project blends sensors, light diffusion, and motor control to create a captivating and immersive experience.

Arduino Project Code 

#include <Servo.h>

//initialize variables
int photo;
int doorOpen = 0;
Servo s;
int btn;
int lights = 0;

void setup() {
  s.attach(9);            // Attach servo to pin 9
  Serial.begin(9600);
  s.write(175);           // Initial door position (closed)

  // Set LED pins as outputs
  pinMode(10, OUTPUT);
  pinMode(11, OUTPUT);
  pinMode(12, OUTPUT);
  pinMode(8, OUTPUT);

  // Set button pin as input with pull-up
  pinMode(4, INPUT_PULLUP);
}

void loop() {
  photo = analogRead(A1);  // Read photoresistor
  Serial.println(photo);

  if (photo > 50) {        // Open door if light threshold is met
    doorOpen = 1;
  }

  if (doorOpen == 1) {
    s.write(55);           // Move servo to open door
  }

  btn = digitalRead(4);    // Check button state

  if (btn == 0) {          // Turn on lights when button pressed
    lights = 1;
  }

  if (lights == 1) {       // Light sequence loop
    int i = 5;
    while (i > 0) {
      digitalWrite(10, HIGH);
      delay(300);
      digitalWrite(10, LOW);
      delay(200);
      digitalWrite(11, LOW);
      delay(500);
      digitalWrite(11, HIGH);
      delay(100);
      digitalWrite(12, HIGH);
      delay(500);
      digitalWrite(8, HIGH);
      delay(300);
      digitalWrite(8, LOW);
      delay(200);
      digitalWrite(12, LOW);
      i--;
    }
  }
}
